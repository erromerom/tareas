Tarea curvas de titulación - complejometría

Primera curva de titulación

Realizar la curva de titulación de una alícuota de 10.00 mL de la mezcla de una solución de nitrato de cobre(II) 0.0100 mol/L y nitrato de cinc 0.0100 y 10.00 mL de solución reguladora de pH 5.50 con EDTA 0.0100 mol/L,

Reacciones

\[\require{mhchem} \begin{array}{rcll} \ce{Cu^{2+} + Y^{4-}} & \ce{<=>} & \ce{CuY^2-} & \log K_{\text{f}} = 18.76\\ \ce{Zn^{2+} + Y^{4-}} & \ce{<=>} & \ce{ZnY^2-} & \log K_{\text{f}} = 16.50\\ \end{array} \]

beta.CuY <- 10^18.76
beta.ZnY <- 10^16.50

Dependencia con el pH del EDTA

\[\require{mhchem} \begin{array}{rcll} \ce{H6Y^2+ + H2O} & \ce{<=>} & \ce{H5Y+ + H3O+} & \text{p}K_{\text{a}1} = 0.0\\ \ce{H5Y^+ + H2O} & \ce{<=>} & \ce{H4Y + H3O+} & \text{p}K_{\text{a}2} = 1.50\\ \ce{H4Y + H2O} & \ce{<=>} & \ce{H3Y^- + H3O+} & \text{p}K_{\text{a}3} = 2.00\\ \ce{H3Y^- + H2O} & \ce{<=>} & \ce{H2Y^2- + H3O+} & \text{p}K_{\text{a}4} = 2.69\\ \ce{H2Y^2- + H2O} & \ce{<=>} & \ce{HY^3- + H3O+} & \text{p}K_{\text{a}5} = 6.13\\ \ce{HY^3- + H2O} & \ce{<=>} & \ce{Y^4- + H3O+} & \text{p}K_{\text{a}6} = 10.37\\ \end{array} \]

alp <- function(pH, pKa){
  n <- length(pKa)
  num <- 10^-(pH * (n:0) + cumsum(c(0,pKa)))
  num / sum(num)
}
pKa <- c(0.0, 1.50, 2.00, 2.69, 6.13, 10.37)
aY <- alp(5.5, pKa)[7]

Balance de masas

\[\require{mhchem} \begin{array}{rll} c(\ce{Cu^2+}) &=\displaystyle 0.0100 \times \cfrac{10.00}{20.00 + V} &= [\ce{Cu^2+}] + [\ce{CuY^2-}]\\ c(\ce{Zn^2+}) &=\displaystyle 0.0100 \times \cfrac{10.00}{20.00 + V} &= [\ce{Zn^2+}] + [\ce{ZnY^2-}]\\ c(\ce{EDTA}) &=\displaystyle 0.0100 \times \cfrac{V}{20.00 + V} &= \cfrac{[\ce{Y^4-}]}{\alpha(\ce{Y^4-})} + [\ce{CuY^2-}] + [\ce{ZnY^2-}]\\ \end{array} \]

\[\require{mhchem} \begin{array}{rl} \alpha(\ce{Cu^2+}) &=\displaystyle \cfrac{1}{1 + K_{\text{f}}(\ce{CuY^2-})\times [\ce{Y^4-}]} \\ \alpha(\ce{Zn^2+}) &=\displaystyle \cfrac{1}{1 + K_{\text{f}}(\ce{ZnY^2-})\times [\ce{Y^4-}]} \\ \end{array} \]

curva <- function(pY, V){
  Y <- 10^-pY
  aCu <- 1 / (1 + beta.CuY * Y)
  aZn <- 1 / (1 + beta.ZnY * Y)
  Cu  <- aCu * 0.1 / (20 + V)
  Zn  <- aZn * 0.1 / (20 + V)
  CuY <- beta.CuY * Cu * Y
  ZnY <- beta.ZnY * Zn * Y
  EDTA <- 0.01 * V / (20 + V)
  EDTA - Y / aY - CuY - ZnY
}

Definiendo la curva de titulación

El volumen se calculará entre 0.05 y 25.00 mL con intervalos de 0.01 mL.

V <- seq(0.05, 25.00, by = 0.01)

Calculando pY y pM

Se calcula el valor de pY para todos los valores de titulante

pY <- sapply(V, function(x)
  uniroot(curva, V = x, interval = c(0, 200), tol = 1e-14)$root)

Con el valor de pY se calcula las concentraciones en equilibrio de los cationes de cobre y cinc.

Y <- 10^-pY
aCu <- 1 / (1 + beta.CuY * Y)
aZn <- 1 / (1 + beta.ZnY * Y)
pCu <- -log10(aCu * 0.1 / (20 + V))
pZn <- -log10(aZn * 0.1 / (20 + V))

La gráfica

se procede hacer la gráfica

if (!require("tidyverse")) install.packages("tidyverse")
df <- data.frame(V, pY, pCu, pZn) %>% 
  pivot_longer(names_to = "key",
               values_to = "y",
               -V)
fig <- ggplot(df,
              aes(x = V, 
                  y = y, 
                  colour = key)
              ) +
  geom_line(size = 1L) +
  scale_color_brewer("", 
                     palette = "Dark2", 
                     direction = 1) +
  labs(x = "Volumen de titulante / mL",
       y = "py o pM") +
  theme_bw() + 
  theme(text = element_text(size = 16),
        legend.position = c(0.15, 0.25))
print(fig)

Realizar la curva de titulación de una alícuota de 10.00 mL de una solución de nitrato de cobre(II) 0.0100 mol/L y 10.00 mL de solución reguladora de pH 10.00 que contiene amoniaco en concentraciones 0.001, 0.01 y 0.1 mol/L (omita el efecto del pH sobre el amoniaco) con EDTA 0.0100 mol/L,

Reacciones

\[\require{mhchem} \begin{array}{rcll} \ce{Cu^{2+} + Y^{4-}} & \ce{<=>} & \ce{CuY^2-} & \log K_{\text{f}} = 18.76\\ \ce{Cu^{2+} + NH3 } & \ce{<=>} & \ce{Cu(NH3)^2+} & \log \beta_{1} = 3.99\\ \ce{Cu^{2+} + 2 NH3 } & \ce{<=>} & \ce{Cu(NH3)2^2+} & \log \beta_{2} = 7.33\\ \ce{Cu^{2+} + 3 NH3 } & \ce{<=>} & \ce{Cu(NH3)3^2+} & \log \beta_{3} = 10.06\\ \ce{Cu^{2+} + 4 NH3 } & \ce{<=>} & \ce{Cu(NH3)4^2+} & \log \beta_{4} = 12.03\\ \end{array} \]

\[\require{mhchem} \begin{array}{rcll} \ce{H6Y^2+ + H2O} & \ce{<=>} & \ce{H5Y+ + H3O+} & \text{p}K_{\text{a}1} = 0.0\\ \ce{H5Y^+ + H2O} & \ce{<=>} & \ce{H4Y + H3O+} & \text{p}K_{\text{a}2} = 1.50\\ \ce{H4Y + H2O} & \ce{<=>} & \ce{H3Y^- + H3O+} & \text{p}K_{\text{a}3} = 2.00\\ \ce{H3Y^- + H2O} & \ce{<=>} & \ce{H2Y^2- + H3O+} & \text{p}K_{\text{a}4} = 2.69\\ \ce{H2Y^2- + H2O} & \ce{<=>} & \ce{HY^3- + H3O+} & \text{p}K_{\text{a}5} = 6.13\\ \ce{HY^3- + H2O} & \ce{<=>} & \ce{Y^4- + H3O+} & \text{p}K_{\text{a}6} = 10.37\\ \end{array} \]

pKa <- c(0.0, 1.50, 2.00, 2.69, 6.13, 10.37)

Balance de masas

\[\require{mhchem} \begin{array}{rll} c(\ce{Cu^2+}) &=\displaystyle 0.0100 \times \cfrac{10.00}{20.00 + V} &=\displaystyle [\ce{Cu^2+}] + \sum_{i=1}^4 [\ce{Cu(NH3)}_i^{2+}] + [\ce{CuY^2-}]\\ c(\ce{EDTA}) &=\displaystyle 0.0100 \times \cfrac{V}{20.00 + V} &= \cfrac{[\ce{Y^4-}]}{\alpha(\ce{Y^4-})} + [\ce{CuY^2-}]\\ \end{array} \]

\[\require{mhchem} \begin{array}{rl} \alpha(\ce{Cu^2+}) &=\displaystyle \cfrac{1}{1 + \displaystyle \sum_{i=1}^4 \beta_i\times [\ce{NH3}]^i + K_{\text{f}}(\ce{CuY^2-})\times [\ce{Y^4-}]} \\ \end{array} \]

curva <- function(pY, V, NH3){
  Y <- 10^-pY
  aCu <- 1 / (1 + beta.CuY * Y + beta.CN1 * NH3 + beta.CN2 * NH3^2 +

  Cu  <- aCu * 0.1 / (20 + V)
  CuY <- beta.CuY * Cu * Y
  EDTA <- 0.01 * V / (20 + V)
  EDTA - Y / aY - CuY
}

Se calcula el valor de pY para todos los valores de titulante

pY <- sapply(CN, function(xx)
  sapply(V, function(x)
  uniroot(curva, V = x, CN = xx, interval = c(0, 200), tol = 1e-14)$root))
pY <- as.data.frame(pY)
names(pY) <- c("CN0000", "CN0001", "CN0010", "CN0100", "CN1000")
pCu <- pY
for (i in 1:5) {
  Y <- 10^-pY[names(pY)[i]]
  aCu <- 1 / (1 + beta.CuY * Y + beta.CN2 * CN[i]^2 +
                beta.CN3 * CN[i]^3 + beta.CN4 * CN[i]^4)
  Cu  <- aCu * 0.1 / (20 + V)
  pCu[names(pCu)[i]] <-  -log10(Cu)
}

Con el valor de pY se calcula las concentraciones en equilibrio de los cationes de cobre y cinc.

df <- data.frame(V, pCu) %>% 
  pivot_longer(values_to = "pCu",
               names_to = "key",
               -V)
